-- Таблица 1: current_traffic (текущий трафик)
CREATE TABLE IF NOT EXISTS mart.current_traffic (
    id SERIAL PRIMARY KEY,
    camera_id VARCHAR(50),
    vehicle_type VARCHAR(50),
    track_id INTEGER,
    x_coord FLOAT,
    y_coord FLOAT,
    avg_speed FLOAT,
    count INTEGER,
    last_update TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Таблица 2: traffic_history (история трафика для аналитики)
CREATE TABLE IF NOT EXISTS mart.traffic_history (
    id SERIAL PRIMARY KEY,
    report_hour TIMESTAMP,
    camera_id VARCHAR(50),
    vehicle_type VARCHAR(50),
    avg_speed FLOAT,
    intensity INTEGER,
    density FLOAT,
    UNIQUE(report_hour, camera_id, vehicle_type)
);

-- Таблица 3: quality_log (логи качества данных)
CREATE TABLE IF NOT EXISTS mart.quality_log (
    id SERIAL PRIMARY KEY,
    issue INTEGER,
    ts TIMESTAMP DEFAULT NOW()
);
-- Создание схемы mart
CREATE SCHEMA IF NOT EXISTS mart;

-- Представление для текущей ситуации
CREATE OR REPLACE VIEW mart.current_situation AS
SELECT 
    camera_id,
    vehicle_type,
    COUNT(DISTINCT track_id) as active_vehicles,
    ROUND(AVG(avg_speed), 1) as avg_speed,
    MAX(avg_speed) as max_speed,
    MIN(avg_speed) as min_speed,
    COUNT(*) as total_detections,
    MAX(last_update) as last_update
FROM mart.current_traffic
WHERE last_update >= NOW() - INTERVAL '5 minutes'
GROUP BY camera_id, vehicle_type;

-- Представление для часовой статистики
CREATE OR REPLACE VIEW mart.hourly_stats AS
SELECT 
    DATE_TRUNC('hour', last_update) as hour,
    camera_id,
    vehicle_type,
    COUNT(DISTINCT track_id) as unique_vehicles,
    AVG(avg_speed) as avg_speed,
    COUNT(*) as detections
FROM mart.current_traffic
WHERE last_update >= NOW() - INTERVAL '24 hours'
GROUP BY DATE_TRUNC('hour', last_update), camera_id, vehicle_type;
