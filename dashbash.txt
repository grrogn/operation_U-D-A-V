def get_raw_frame(cap):
    success, frame = cap.read()
    if not success:
        cap.set(cv2.CAP_PROP_POS_FRAMES, 0)
        return None
    return frame
def process_and_draw(frame, model, track_history, font):
    results = model.track(frame, persist=True, classes=[2, 5, 7],
                          conf=0.3, iou=0.5, imgsz=1280, verbose=False)
    db_rows = []
    if results[0].boxes.id is not None:
        boxes = results[0].boxes.xyxy.int().cpu().numpy()
        ids = results[0].boxes.id.int().cpu().tolist()
        clss = results[0].boxes.cls.int().cpu().tolist()
        centers = results[0].boxes.xywh.cpu().numpy()
        img_pil = Image.fromarray(cv2.cvtColor(frame, cv2.COLOR_BGR2RGB))
        draw = ImageDraw.Draw(img_pil)
        for box, track_id, cls, center in zip(boxes, ids, clss, centers):
            x1, y1 = box[0], box[1]
            x_c, y_c = int(center[0]), int(center[1])
            label = model.names[cls].upper()
            speed = round(abs(y_c - track_history.get(track_id, y_c)) * 2.5, 1)
            track_history[track_id] = y_c
            db_rows.append(('Cam_01', label, int(track_id), float(x_c), float(y_c), float(speed)))
            time_str = datetime.datetime.now().strftime("%H:%M:%S")
            display_text = f" {label} ID:{track_id} | {time_str} | {speed} km/h "
            tw, th = draw.textbbox((0, 0), display_text, font=font)[2:]
            draw.rectangle([x1, y1 - th - 10, x1 + tw, y1], fill=(255, 100, 0))
            draw.text((x1, y1 - th - 7), display_text, font=font, fill=(255, 255, 255))
        frame = cv2.cvtColor(np.array(img_pil), cv2.COLOR_RGB2BGR)
    return db_rows, frame
def push_to_database(cur, conn, data_list):
    for row in data_list:
        try:
            cur.execute("""
                INSERT INTO mart.current_traffic 
                (camera_id, vehicle_type, track_id, x_coord, y_coord, avg_speed, count, last_update) 
                VALUES (%s, %s, %s, %s, %s, %s, %s, CURRENT_TIMESTAMP)
            """, (row[0], row[1], row[2], row[3], row[4], row[5], 1))
            conn.commit()
        except:
            conn.rollback()
def run_streaming_pipeline():
    while True:
        try:
            conn = psycopg2.connect(**db_config)
            cur = conn.cursor()
            cur.execute("""
                INSERT INTO mart.traffic_history (report_hour, camera_id, vehicle_type, avg_speed, intensity, density)
                SELECT date_trunc('hour', last_update), camera_id, vehicle_type, AVG(avg_speed), COUNT(*), AVG(count)
                FROM mart.current_traffic GROUP BY 1, 2, 3
                ON CONFLICT (report_hour, camera_id, vehicle_type) 
                DO UPDATE SET avg_speed = EXCLUDED.avg_speed, intensity = EXCLUDED.intensity, density = EXCLUDED.density;
            """)
            conn.commit()
            cur.close()
            conn.close()
            time.sleep(5)
        except:
            time.sleep(5)
def process_video():
    model = YOLO(MODEL_NAME)
    cap = cv2.VideoCapture("test.MP4")
    track_history = {}
    try:
        font = ImageFont.truetype("arial.ttf", 20)
    except:
        font = ImageFont.load_default()
    while cap.isOpened():
        conn = psycopg2.connect(**db_config)
        cur = conn.cursor()
        frame = get_raw_frame(cap)
        if frame is not None:
            db_data, frame = process_and_draw(frame, model, track_history, font)
            if db_data:
                push_to_database(cur, conn, db_data)
        cur.close()
        conn.close()
if 'video_thread_started' not in st.session_state:
    threading.Thread(target=process_video, daemon=True).start()
    threading.Thread(target=run_streaming_pipeline, daemon=True).start()
    st.session_state.video_thread_started = True

@st.cache_data(ttl=5)
def load_data():
    conn = psycopg2.connect(**db_config)
    df = pd.read_sql("SELECT * FROM mart.traffic_history ORDER BY report_hour DESC LIMIT 1000", conn)
    conn.close()
    return df
@st.cache_data(ttl=10)
def load_current_data():
    try:
        conn = psycopg2.connect(**db_config)
        df = pd.read_sql("""
            SELECT camera_id, vehicle_type, AVG(avg_speed) as avg_speed, COUNT(*) as intensity
            FROM mart.current_traffic
            WHERE last_update > NOW() - INTERVAL '5 minutes'
            GROUP BY camera_id, vehicle_type
        """, conn)
        conn.close()
        return df
    except:
        return pd.DataFrame()
try:
    df = load_data()
    if df.empty:
        st.info("Ожидание данных...")
    else:
        df['report_hour'] = pd.to_datetime(df['report_hour'])
        st.sidebar.header("Фильтры")
        selected_camera = st.sidebar.selectbox("Камера", df['camera_id'].unique())
        selected_types = st.sidebar.multiselect("Типы ТС", df['vehicle_type'].unique(),
                                                default=df['vehicle_type'].unique())
        hours = st.sidebar.slider("Часы суток", 0, 23, (0, 23))
        df_filtered = df[
            (df['camera_id'] == selected_camera) & 
            (df['vehicle_type'].isin(selected_types)) &
            (df['report_hour'].dt.hour >= hours[0]) &
            (df['report_hour'].dt.hour <= hours[1])
        ]
        if not df_filtered.empty:
            last = df_filtered.iloc[0]
            current_df = load_current_data()
            if not current_df.empty:
                current_filtered = current_df[current_df['camera_id'] == selected_camera]
                if not current_filtered.empty:
                    st.subheader("📊 Текущее состояние")
                    col1, col2, col3 = st.columns(3)
                    col1.metric("Средняя скорость", f"{current_filtered['avg_speed'].mean():.0f} км/ч")
                    col2.metric("Интенсивность", f"{int(current_filtered['intensity'].sum())} ТС/мин")
                    col3.metric("Активных ТС", f"{len(current_filtered)}")
            st.subheader("📊 Текущее состояние и прогноз")
            c1, c2, c3, c4, c5 = st.columns(5)
            c1.metric("Интенсивность", f"{int(last['intensity'])} ТС/ч")
            c2.metric("Ср. Скорость", f"{last['avg_speed']:.0f} км/ч")
            c3.metric("Прогноз 30м", f"{int(last['intensity'] * 1.15)} ТС")
            c4.metric("Прогноз 60м", f"{int(last['intensity'] * 1.25)} ТС")
            c5.metric("Прогноз 120м", f"{int(last['intensity'] * 1.35)} ТС")
            st.subheader("📊 Сравнение с прошлыми периодами")
            today_data = df_filtered[df_filtered['report_hour'].dt.date == datetime.now().date()]
            yesterday_data = df_filtered[df_filtered['report_hour'].dt.date == (datetime.now() - timedelta(days=1)).date()]
            week_data = df_filtered[df_filtered['report_hour'].dt.date >= (datetime.now() - timedelta(days=7)).date()]
            col1, col2, col3 = st.columns(3)
            if not today_data.empty:
                col1.metric("Сегодня", f"{int(today_data['intensity'].sum())} ТС")
            else:
                col1.metric("Сегодня", "Нет данных")
            if not yesterday_data.empty:
                col2.metric("Вчера", f"{int(yesterday_data['intensity'].sum())} ТС")
            else:
                col2.metric("Вчера", "Нет данных")
            if not week_data.empty:
                col3.metric("За неделю", f"{int(week_data['intensity'].sum())} ТС")
            else:
                col3.metric("За неделю", "Нет данных")
            st.subheader("📉 Анализ трафика и структуры")
            col1, col2 = st.columns(2)
            hourly = df_filtered.groupby(df_filtered['report_hour'].dt.hour)['intensity'].mean().reset_index()
            hourly.columns = ['часы', 'Интенсивность']
            col1.plotly_chart(px.line(hourly, x='часы', y='Интенсивность', title="Загруженность по часам"),
                              use_container_width=True)
            peak_hours = hourly.nlargest(3, 'Интенсивность')['часы'].tolist()
            col1.info(f"🚨 Часы пик: {', '.join([f'{int(h)}:00' for h in peak_hours])}")
            
            structure = df_filtered.groupby('vehicle_type')['intensity'].sum().reset_index()
            col2.plotly_chart(
                px.pie(structure, values='intensity', names='vehicle_type', title="Доли транспорта", hole=0.4),
                use_container_width=True)
            st.subheader("📈 Оценка эффективности модели прогнозирования")

            if len(full_df) > 10:

                train = full_df.iloc[:int(len(full_df) * 0.8)]
                test = full_df.iloc[int(len(full_df) * 0.8):]
                model = GradientBoostingRegressor().fit(train[['hour']], train['intensity'])

                    os.makedirs('models', exist_ok=True)

                except:
                    joblib.dump(model, 'traffic_model.pkl')

                col1.metric("MAE", f"{mean_absolute_error(test['intensity'], pred):.1f}")
                col2.metric("RMSE", f"{np.sqrt(mean_squared_error(test['intensity'], pred)):.1f}")
